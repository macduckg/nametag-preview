    <script>
        // 요소 가져오기
        const textInput = document.getElementById('textInput');
        const previewText = document.getElementById('previewText');
        const previewBox = document.getElementById('previewBox');
        const fontSelect = document.getElementById('fontSelect');
        const bgBtns = document.querySelectorAll('#bgColorOptions .color-btn');
        const textBtns = document.querySelectorAll('#textColorOptions .color-btn');

        // 1. 텍스트 변경
        textInput.addEventListener('input', (e) => {
            previewText.innerText = e.target.value || 'NAMETAG';
        });

        // 2. 폰트 변경
        fontSelect.addEventListener('change', (e) => {
            previewText.style.fontFamily = e.target.value;
        });

        // 3. 배경 색상 변경
        bgBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                // 기존 선택 해제 및 새 선택 표시
                bgBtns.forEach(b => b.classList.remove('selected'));
                btn.classList.add('selected');
                // 색상 적용
                previewBox.style.backgroundColor = btn.getAttribute('data-color');
            });
        });

        // 4. 글자 색상 변경
        textBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                textBtns.forEach(b => b.classList.remove('selected'));
                btn.classList.add('selected');
                previewText.style.color = btn.getAttribute('data-color');
            });
        });
    </script>
